### 通过工厂模式创建
```function Human(){
			this.name = '张三';
			this.age=20;
			this.say=function(){
				alert(this.name+"de 年龄为："+this.age);
			}		
		}

		var zs =new Human();
		alert('手动弹出name：'+zs.name);

		var xz =new Human();
		alert( '第二次手动弹出：'+xz.name);

		zs.say();
```
